from simple_markov_chain_lib import markov_chain

## Build Markov Chain
markov_table = {
    1: {2: 0.5, 3: 0.5},
    2: {1: 1/3, 4: 2/3},
    3: {3: 0.8, 4: 0.2},
    4: {1: 0.6, 4: 0.4}
}

init_dist = {1: 1.0} #the chain starts from state 1 since we are interested in excursions around state 1.

mc = markov_chain(markov_table, init_dist)

N = 1000

##Initialize the dictionary "visits". Every state is a key of this dictionary, with value 0
visits = {state: 0 for state in (1, 2, 3, 4)} 

mc.start()
completed = 0
time_counter = 0

##Simulate N excursions. Every time we complete an excursion we increase completed by 1.
while completed < N:
    time_counter += 1
    mc.move()
    if mc.running_state == 1:
        completed += 1

print("Expected Time to come back to 1:")
print("%.3f" % (time_counter / N))

print("Distribution of 1:")
print("%.3f" % (N / time_counter))

Υπολογίζουμε θεωρητικά πως η τιμή της $π_1$ είναι $18/97$, άρα η εκτιμήτρια μας είναι πολύ κοντά. Για τον υπολογισμό της τιμής της αναλλοίωτης κατανομής στις άλλες καταστάσεις, αρκεί να αλλάξουμε την αρχική κατάσταση, καθώς και την συνθήκη του if statement στο while loop.

from numpy import random, pi

Ntrials, Nhits = 1_000_000, 0
for n in range(Ntrials):
    x, y, z = random.uniform(-1, 1, 3) # draw 3 samples, each uniformly distributed over (-1,1)
    if x**2 + y**2 + z**2 < 1:
        Nhits += 1

print("Monte Carlo estimator of Sphere's volume: %.5f" % (( Nhits / Ntrials ) * 8 ))
print("Actual value of Sphere up to 5 decimal digits: %.5f" % ( 4 * pi / 3 ))
print("The relative error is %.5f%%" % (100 * abs(1- (8 * Nhits)/((4 * pi * Ntrials) / 3))))

# Parameters
N = 10_000  # number of required points
Ntrials, Nhits = 0, 0
acc_x, acc_y = [], []  # accepted x, y
rej_x, rej_y = [], []  # rejected x, y

# Rejection Sampling
while Nhits < N:
    Ntrials += 1
    x, y = random.uniform(-1, 1, 2)
    if (x**2 + y**2)**2 < 2*abs(x)*abs(y):
        acc_x.append(x)
        acc_y.append(y)
        Nhits += 1
    else:
        rej_x.append(x)
        rej_y.append(y)

print("Total number of samples drawn %d" % Ntrials)
print("Number of samples in the pool %d" % N)


import matplotlib.pyplot as plt

fig, ax = plt.subplots()

plt.scatter(acc_x, acc_y, color = 'green', s = 1) # parameter s determines the size of each dot in the scatter plot
plt.scatter(rej_x, rej_y, color = 'red', s = 1)

ax.set_xlim([-1, 1])
ax.set_ylim([-1, 1])
ax.set_aspect('equal')  # set aspect ratio 1:1

plt.show()
